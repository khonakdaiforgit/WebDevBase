    /// <summary>

namespace MyApp.Application.Restaurants
{
    // ----- DTOهای ورودی -----
    public record CreateRestaurantDto(
        string Name,
        string Address,
        string Phone,
        string Email,
        string LogoUrl,
        WorkingHours WorkingHours);

    public record UpdateRestaurantDto(
        Guid Id,
        string Name,
        string Address,
        string Phone,
        string Email,
        string? LogoUrl,
        WorkingHours? WorkingHours);

    public record UpdateLogoDto(Guid RestaurantId, string LogoUrl);
    public record UpdateWorkingHoursDto(Guid RestaurantId, WorkingHours WorkingHours);

    // ----- DTOهای خروجی -----
    public record RestaurantDto(
        Guid Id,
        string Name,
        string Address,
        string Phone,
        string Email,
        string LogoUrl,
        WorkingHours WorkingHours);

    public record RestaurantListItemDto(
        Guid Id,
        string Name,
        string Address,
        string LogoUrl);

    /// <summary>
    /// سرویس مدیریت رستوران‌ها
    /// </summary>
    public interface IRestaurantService
    {
        Task<Guid> CreateAsync(CreateRestaurantDto dto, Guid ownerUserId);
        Task UpdateAsync(UpdateRestaurantDto dto, Guid callerUserId);
        Task DeleteAsync(Guid restaurantId, Guid callerUserId);
        Task<RestaurantDto?> GetAsync(Guid restaurantId);
        Task<PagedResult<RestaurantListItemDto>> GetListAsync(int page = 1, int pageSize = 20, Guid? ownerUserId = null);

        Task UpdateLogoAsync(UpdateLogoDto dto, Guid callerUserId);
        Task UpdateWorkingHoursAsync(UpdateWorkingHoursDto dto, Guid callerUserId);
    }
}

namespace MyApp.Application.Authorization
{
    /// <summary>
    /// نقش‌های مجاز برای هر عملیات
    /// </summary>
    public enum AccessLevel
    {
        Owner,      // مالک پروژه (فقط شما)
        StoreAdmin  // مدیر فروشگاه (Restaurant Owner)
    }

    public interface IAuthorizationService
    {
        Task<bool> CanAccessRestaurantAsync(Guid userId, Guid restaurantId, AccessLevel required = AccessLevel.StoreAdmin);
        Task<bool> CanViewLogsAsync(Guid userId); // فقط Owner
        Task<bool> IsOwnerAsync(Guid userId);
    }
}

// 12. واحد کار (Unit of Work) – برای تراکنش‌ها

namespace MyApp.Application.Common
{
    public interface IUnitOfWork : IDisposable
    {
        Task<int> SaveChangesAsync(CancellationToken cancellationToken = default);

        Task RollbackTransactionAsync();
    }
}

// نکات مهم پیاده‌سازی بعدی (بدون کد):
// 1. تمام سرویس‌ها از IUnitOfWork استفاده می‌کنند.
// 2. دسترسی به رستوران فقط با بررسی مالکیت (Restaurant.OwnerUserId == callerId یا Owner پروژه).
// 3. لاگ‌ها فقط برای Owner قابل مشاهده → IAuthorizationService.CanViewLogsAsync
// 4. MongoDB → Repositoryهای ژنریک با IQueryable برای فیلتر و صفحه‌بندی.
// 5. DTOها در Presentation (MVC یا API) به مدل‌های View تبدیل می‌شوند.
